  <script>
    function checkForm() {
      let form = document.forms["signup"];
      let first = form["firstname"].value.trim();
      let last = form["lastname"].value.trim();
      let pass1 = form["password"].value;
      let pass2 = form["confirm"].value;
      let mail = form["useremail"].value.trim();
      let mobile = form["phone"].value.trim();

      // First name check
      if (first.length < 6) {
        alert("First name must be at least 6 characters long");
        return false;
      }
      if (/[^a-zA-Z0-9]/.test(first)) {
        alert("First name cannot contain special characters");
        return false;
      }

      // Password check
      if (pass1.length < 6) {
        alert("Password must have 6 or more characters");
        return false;
      }
      if (pass1 !== pass2) {
        alert("Passwords are not the same");
        return false;
      }

      // Email check
      let mailRegex = /^[^@\s]+@[^@\s]+\.[^@\s]+$/;
      if (!mailRegex.test(mail)) {
        alert("Please enter a valid email address");
        return false;
      }

      // Phone number check
      if (!/^[0-9]{10}$/.test(mobile)) {
        alert("Phone number must be 10 digits only");
        return false;
      }

      alert("Registration successful!");
      return true;
    }
  </script>
